Demo of auto-completion system with autoCompltr.js

Choose a Surname (ex : Rose)

Projet réalisé par Jérémie Boulay

AutoCompltr.js is a simple, basic and fast system of auto-completion. I wanted to have a system of auto-completion without deppendencies and quickly usable.

Script Written

			var autoCompltrDOM = document.getElementById('auto-complete');
			var list = ["Hewitt","Carter","Murray","Bartlett","Howard","Lancaster","Steele","..."];

			var autoCompltr = new AutoCompltr(autoCompltrDOM, list);

			autoCompltr.onEnter(function(e){
				alert("You choose : "+autoCompltr.getValue()+" with the enter key");
			},false);

			document.getElementById('show-surname').addEventListener('click', function(e){
				alert("You choose : "+autoCompltr.getValue());
			});
